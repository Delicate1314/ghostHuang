| 模块 | 选型 | 理由 |
|------|------|------|
| 数据集 | MovieLens 1M（原型）→ MovieLens 25M（正式） | 1M 只有 3,706 部电影，适合快速验证；25M 有 62,423 部，适合出论文 |
| 元数据 | TMDb API | 免费、字段全、有海报链接 |
| LLM | **DeepSeek-V3.2**（`deepseek-chat` API） | 128K 上下文、支持 JSON Output、极低成本（¥2/百万输入 token） |
| LLM 后训练 | DeepSeek 微调 API + 人工校验数据 | 云端微调，不占本地显存 |

| 对齐训练 | 对比学习（InfoNCE）将 LLM/VLM 表示与协同过滤信号对齐（本地训练） | 轻量、有效、有理论支撑 |
| 语义 ID | 显式语义 token + RQ-VAE 隐式量化 token | 混合型 SID 是当前 SOTA 思路 |
| 推荐模型 | Transformer Decoder（类 TIGER 架构，本地训练） | 生成式推荐的标准范式 |
| 文本 Embedding | **BGE-M3**（本地） | 小模型，~600MB 显存，4060 轻松跑 |
| 基线框架 | RecBole（本地） | SASRec/GRU4Rec/LightGCN 等基线一键跑 |
| 硬件 | **1× RTX 4060 8GB** | CLIP + 对齐训练 + 推荐模型均可本地跑，LLM/VLM 走 API |

### 2.2 从 TMDb 获取元数据

```python
# src/data/fetch_tmdb.py
import requests, json, time, os
import pandas as pd

TMDB_API_KEY = "你的API Key"  # 去 https://www.themoviedb.org/settings/api 免费申请
BASE_URL = "https://api.themoviedb.org/3"

def fetch_movie_details(tmdb_id: int) -> dict:
    """获取单部电影的详细信息"""
    url = f"{BASE_URL}/movie/{tmdb_id}"
    params = {
        "api_key": TMDB_API_KEY,
        "language": "en-US",
        "append_to_response": "credits"
    }
    resp = requests.get(url, params=params, timeout=10)
    if resp.status_code != 200:
        return None
    data = resp.json()
    return {
        "tmdb_id": tmdb_id,
        "title": data.get("title"),
        "overview": data.get("overview", ""),
        "genres": [g["name"] for g in data.get("genres", [])],
        "release_date": data.get("release_date", ""),
        "vote_average": data.get("vote_average", 0),
        "poster_path": data.get("poster_path"),
        "directors": [c["name"] for c in data.get("credits", {}).get("crew", []) 
                       if c["job"] == "Director"],
        "cast": [c["name"] for c in data.get("credits", {}).get("cast", [])[:10]],
    }

def batch_fetch(links_csv: str, output_path: str, cache_path: str = "data/raw/tmdb_cache.json"):
    """批量获取，支持断点续传"""
    links = pd.read_csv(links_csv)
    
    # 加载已缓存
    cache = {}
    if os.path.exists(cache_path):
        with open(cache_path, 'r') as f:
            cache = json.load(f)
    
    results = []
    for _, row in links.iterrows():
        tmdb_id = str(int(row["tmdbId"])) if pd.notna(row["tmdbId"]) else None
        if tmdb_id is None:
            continue
        if tmdb_id in cache:
            results.append(cache[tmdb_id])
            continue
        
        detail = fetch_movie_details(int(tmdb_id))
        if detail:
            cache[tmdb_id] = detail
            results.append(detail)
        
        time.sleep(0.05)  # 控制速率，约 20 req/s
        
        # 每 500 条存一次缓存
        if len(results) % 500 == 0:
            with open(cache_path, 'w') as f:
                json.dump(cache, f)
            print(f"已获取 {len(results)} 部电影")
    
    # 最终保存
    with open(cache_path, 'w') as f:
        json.dump(cache, f)
    pd.DataFrame(results).to_parquet(output_path)
    print(f"完成！共 {len(results)} 部电影 → {output_path}")
```

# src/data/download_posters.py
import requests, os
from concurrent.futures import ThreadPoolExecutor
import pandas as pd

POSTER_BASE = "https://image.tmdb.org/t/p/w500"

def download_one(args):
    tmdb_id, poster_path, save_dir = args
    if poster_path is None:
        return
    url = f"{POSTER_BASE}{poster_path}"
    save_path = os.path.join(save_dir, f"{tmdb_id}.jpg")
    if os.path.exists(save_path):
        return
    try:
        resp = requests.get(url, timeout=15)
        if resp.status_code == 200:
            with open(save_path, 'wb') as f:
                f.write(resp.content)
    except Exception:
        pass

def batch_download(meta_path: str, save_dir: str = "data/raw/posters", workers: int = 8):
    os.makedirs(save_dir, exist_ok=True)
    df = pd.read_parquet(meta_path)
    tasks = [(row["tmdb_id"], row["poster_path"], save_dir) for _, row in df.iterrows()]
    
    with ThreadPoolExecutor(max_workers=workers) as pool:
        list(pool.map(download_one, tasks))
    
    downloaded = len(os.listdir(save_dir))
    print(f"海报下载完成：{downloaded}/{len(tasks)}")
```

### 2.4 构建用户行为序列

```python
# src/data/build_sequences.py
import pandas as pd
import numpy as np

def build_sequences(ratings_path: str, min_len: int = 5, max_len: int = 50):
    """
    按时间排序，构建用户观影序列
    划分方式：leave-last-one-out
    """
    if ratings_path.endswith(".dat"):
        # ML-1M 格式
        df = pd.read_csv(ratings_path, sep="::", header=None,
                         names=["userId", "movieId", "rating", "timestamp"],
                         engine="python")
    else:
        df = pd.read_csv(ratings_path)
    
    # 只保留正反馈（rating >= 4）
    df = df[df["rating"] >= 4].copy()
    df = df.sort_values(["userId", "timestamp"])
    
    # 构建序列
    sequences = df.groupby("userId")["movieId"].apply(list).reset_index()
    sequences.columns = ["userId", "movie_seq"]
    
    # 过滤过短的序列
    sequences = sequences[sequences["movie_seq"].apply(len) >= min_len]
    
    # 截断过长的序列（只保留最近 max_len 部）
    sequences["movie_seq"] = sequences["movie_seq"].apply(lambda x: x[-max_len:])
    
    # Leave-last-one-out 划分
    train, val, test = [], [], []
    for _, row in sequences.iterrows():
        uid = row["userId"]
        seq = row["movie_seq"]
        train.append({"userId": uid, "movie_seq": seq[:-2]})
        val.append({"userId": uid, "movie_seq": seq[:-1], "target": seq[-2]})
        test.append({"userId": uid, "movie_seq": seq[:-1], "target": seq[-1]})
    
    return pd.DataFrame(train), pd.DataFrame(val), pd.DataFrame(test)
```

### 2.5 冷启动划分

```python
# src/data/cold_start_split.py
import numpy as np
import pandas as pd

def make_cold_start_split(meta_df: pd.DataFrame, interaction_df: pd.DataFrame, 
                          cold_ratio: float = 0.15, seed: int = 42):
    """
    随机选 15% 的电影作为冷启动电影：
    - 从训练集中移除它们的所有交互
    - 测试时评估这些电影能否被推荐出来
    """
    np.random.seed(seed)
    all_movies = meta_df["movieId"].unique()
    n_cold = int(len(all_movies) * cold_ratio)
    cold_movies = set(np.random.choice(all_movies, n_cold, replace=False))
    
    # 冷启动电影只保留元数据（简介、海报），不保留任何交互
    warm_interactions = interaction_df[~interaction_df["movieId"].isin(cold_movies)]
    cold_test_pool = meta_df[meta_df["movieId"].isin(cold_movies)]
    
    print(f"冷启动电影数：{len(cold_movies)}/{len(all_movies)}")
    print(f"暖电影交互数：{len(warm_interactions)}/{len(interaction_df)}")
    
    return warm_interactions, cold_movies, cold_test_pool
```

### 3.3 批量提取

```python
# src/llm/extract_semantics.py
import json, os, time
from openai import OpenAI
import pandas as pd
from tqdm import tqdm

def get_client():
    return OpenAI(
        base_url="https://api.deepseek.com",
        api_key=os.environ.get("DEEPSEEK_API_KEY", "你的API Key"),
    )

def extract_one(client: OpenAI, movie: dict) -> dict:
    from .prompts import SEMANTIC_EXTRACTION_PROMPT
    prompt = SEMANTIC_EXTRACTION_PROMPT.format(
        title=movie.get("title", ""),
        overview=movie.get("overview", ""),
        genres=", ".join(movie.get("genres", [])),
        director=", ".join(movie.get("directors", [])),
        cast=", ".join(movie.get("cast", [])[:5]),
        year=movie.get("release_date", "")[:4],
    )
    
    response = client.chat.completions.create(
        model="deepseek-chat",
        messages=[{"role": "user", "content": prompt}],
        temperature=0.1,
        max_tokens=512,
        response_format={"type": "json_object"},  # DeepSeek 原生支持 JSON 输出
    )
    
    text = response.choices[0].message.content.strip()
    try:
        return json.loads(text)
    except json.JSONDecodeError:
        return None

def batch_extract(meta_path: str, output_path: str, cache_path: str = "data/processed/llm_cache.json"):
    client = get_client()
    df = pd.read_parquet(meta_path)
    
    cache = {}
    if os.path.exists(cache_path):
        with open(cache_path, 'r') as f:
            cache = json.load(f)
    
    results = {}
    failed = 0
    for _, row in tqdm(df.iterrows(), total=len(df)):
        mid = str(row["tmdb_id"])
        if mid in cache:
            results[mid] = cache[mid]
            continue
        
        try:
            tags = extract_one(client, row.to_dict())
            if tags:
                results[mid] = tags
                cache[mid] = tags
            else:
                failed += 1
        except Exception as e:
            failed += 1
            print(f"电影 {mid} 失败: {e}")
            time.sleep(1)  # 遇到限流等一秒
        
        if len(results) % 200 == 0:
            with open(cache_path, 'w') as f:
                json.dump(cache, f, ensure_ascii=False)
    
    with open(output_path, 'w', encoding='utf-8') as f:
        json.dump(results, f, ensure_ascii=False, indent=2)
    print(f"语义标签提取完成：成功 {len(results)} 部，失败 {failed} 部")
```

### 3.4 LLM 后训练（DeepSeek 微调 API）— 课题要求的"后训练"

**为什么要后训练？**  
通用 LLM 对"推荐语义"的理解不够精准。例如它可能不知道 "mood=dark" 和 "mood=tense" 在推荐场景下的区别。通过在电影领域数据上做微调，让 LLM 输出更一致、更贴合推荐任务的语义标签。

**DeepSeek 微调 API 流程：**

```python
# src/llm/finetune_deepseek.py
"""
通过 DeepSeek 微调 API 进行后训练
流程：
1. 先用零样本 prompt 对 500 部知名电影提取标签
2. 人工校正/筛选出 300-500 条高质量 (input, output) 对
3. 转换为 JSONL 格式上传
4. 调用微调 API 训练
5. 用微调后的模型重新提取全部标签
"""
import json, os
from openai import OpenAI

def get_client():
    return OpenAI(
        base_url="https://api.deepseek.com",
        api_key=os.environ.get("DEEPSEEK_API_KEY", "你的API Key"),
    )

# ========== Step 1: 准备训练数据 ==========

def prepare_finetune_data(verified_tags_path: str, output_jsonl: str):
    """
    将人工校验的高质量标签转为 DeepSeek 微调所需的 JSONL 格式
    
    每行格式：
    {"messages": [{"role": "system", "content": "..."}, {"role": "user", "content": "..."}, {"role": "assistant", "content": "..."}]}
    """
    with open(verified_tags_path, 'r', encoding='utf-8') as f:
        data = json.load(f)
    
    system_prompt = (
        "You are a movie semantic analyzer for a recommendation system. "
        "Extract structured semantic tags from movie information. "
        "Always output valid JSON with consistent tag values."
    )
    
    with open(output_jsonl, 'w', encoding='utf-8') as f:
        for movie_id, item in data.items():
            example = {
                "messages": [
                    {"role": "system", "content": system_prompt},
                    {"role": "user", "content": item["input"]},
                    {"role": "assistant", "content": json.dumps(item["output"], ensure_ascii=False)}
                ]
            }
            f.write(json.dumps(example, ensure_ascii=False) + "\n")
    
    print(f"微调数据已生成：{output_jsonl}，共 {len(data)} 条")


# ========== Step 2: 上传数据并创建微调任务 ==========

def create_finetune_job(training_file_path: str):
    """
    上传训练数据并启动微调
    
    注意：DeepSeek 微调 API 的具体接口以官方文档为准
    https://platform.deepseek.com/docs
    """
    client = get_client()
    
    # 上传训练文件
    with open(training_file_path, "rb") as f:
        file_response = client.files.create(file=f, purpose="fine-tune")
    file_id = file_response.id
    print(f"文件已上传：{file_id}")
    
    # 创建微调任务
    job = client.fine_tuning.jobs.create(
        training_file=file_id,
        model="deepseek-chat",
        hyperparameters={
            "n_epochs": 3,
        }
    )
    print(f"微调任务已创建：{job.id}")
    print(f"状态：{job.status}")
    return job.id


def check_finetune_status(job_id: str):
    """查看微调任务状态"""
    client = get_client()
    job = client.fine_tuning.jobs.retrieve(job_id)
    print(f"任务 {job_id}:")
    print(f"  状态：{job.status}")
    print(f"  微调模型：{job.fine_tuned_model}")
    return job


# ========== Step 3: 使用微调后的模型 ==========

def extract_with_finetuned(movie: dict, finetuned_model: str) -> dict:
    """用微调后的模型提取语义标签"""
    client = get_client()
    from .prompts import SEMANTIC_EXTRACTION_PROMPT
    
    prompt = SEMANTIC_EXTRACTION_PROMPT.format(
        title=movie.get("title", ""),
        overview=movie.get("overview", ""),
        genres=", ".join(movie.get("genres", [])),
        director=", ".join(movie.get("directors", [])),
        cast=", ".join(movie.get("cast", [])[:5]),
        year=movie.get("release_date", "")[:4],
    )
    
    response = client.chat.completions.create(
        model=finetuned_model,  # 使用微调后的模型名称
        messages=[
            {"role": "system", "content": "You are a movie semantic analyzer for a recommendation system."},
            {"role": "user", "content": prompt}
        ],
        temperature=0.1,
        max_tokens=512,
        response_format={"type": "json_object"},
    )
    
    return json.loads(response.choices[0].message.content)

import torch
import numpy as np
import open_clip
from PIL import Image
from pathlib import Path
from tqdm import tqdm

def extract_clip_features(poster_dir: str, output_path: str, batch_size: int = 32):
    model, _, preprocess = open_clip.create_model_and_transforms(
        'ViT-L-14', pretrained='openai'
    )
    model = model.cuda().eval()
    
    poster_files = sorted(Path(poster_dir).glob("*.jpg"))
    
    all_features = []
    all_ids = []
    
    for i in tqdm(range(0, len(poster_files), batch_size)):
        batch_files = poster_files[i:i+batch_size]
        images = []
        ids = []
        for f in batch_files:
            try:
                img = preprocess(Image.open(f).convert("RGB"))
                images.append(img)
                ids.append(f.stem)  # tmdb_id
            except Exception:
                continue
        
        if not images:
            continue
        
        batch = torch.stack(images).cuda()
        with torch.no_grad():
            features = model.encode_image(batch)
            features = features / features.norm(dim=-1, keepdim=True)  # L2 归一化
        
        all_features.append(features.cpu().numpy())
        all_ids.extend(ids)
    
    all_features = np.concatenate(all_features, axis=0)  # [N, 768]
    np.save(output_path, all_features)
    
    import json
    with open(output_path.replace(".npy", "_ids.json"), 'w') as f:
        json.dump(all_ids, f)
    
    print(f"CLIP 特征提取完成：{all_features.shape}")
    return all_features, all_ids

### 4.3 多模态对齐训练 — 课题要求的"对齐"

**核心思想：** LLM 的文本 embedding 和 CLIP 的视觉 embedding 处于不同语义空间。同时，它们都不直接反映推荐信号（哪些电影会被同一个用户喜欢）。对齐训练的目标是：

1. 将文本和视觉表示投影到**同一空间**
2. 让这个空间同时反映**协同过滤信号**（共现关系）

```python
# src/vlm/align_features.py
"""
对比学习对齐：
- 正样本：同一用户喜欢的电影对
- 负样本：随机电影对
"""
import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np

class AlignmentModel(nn.Module):
    """将文本 embedding 和视觉 embedding 投影到共同空间"""
    def __init__(self, text_dim=1024, visual_dim=768, proj_dim=256):
        super().__init__()
        self.text_proj = nn.Sequential(
            nn.Linear(text_dim, 512),
            nn.GELU(),
            nn.Linear(512, proj_dim),
        )
        self.visual_proj = nn.Sequential(
            nn.Linear(visual_dim, 512),
            nn.GELU(),
            nn.Linear(512, proj_dim),
        )
        self.temperature = nn.Parameter(torch.tensor(0.07))
    
    def forward(self, text_emb, visual_emb):
        text_z = F.normalize(self.text_proj(text_emb), dim=-1)
        visual_z = F.normalize(self.visual_proj(visual_emb), dim=-1)
        return text_z, visual_z
    
    def get_fused(self, text_emb, visual_emb, alpha=0.6):
        """融合后的多模态表示"""
        text_z, visual_z = self.forward(text_emb, visual_emb)
        return F.normalize(alpha * text_z + (1 - alpha) * visual_z, dim=-1)


def info_nce_loss(z_i, z_j, temperature=0.07):
    """InfoNCE 对比损失"""
    batch_size = z_i.shape[0]
    z = torch.cat([z_i, z_j], dim=0)  # [2B, D]
    sim = torch.mm(z, z.t()) / temperature  # [2B, 2B]
    
    # 正样本 mask
    labels = torch.arange(batch_size, device=z.device)
    labels = torch.cat([labels + batch_size, labels])  # 正样本索引
    
    # 去掉自身
    mask = ~torch.eye(2 * batch_size, dtype=torch.bool, device=z.device)
    sim = sim.masked_select(mask).view(2 * batch_size, -1)
    
    # 调整 labels
    loss = F.cross_entropy(sim, labels)
    return loss


def train_alignment(text_embeddings: np.ndarray, visual_embeddings: np.ndarray,
                    cooccurrence_pairs: list, epochs: int = 50, batch_size: int = 256):
    """
    cooccurrence_pairs: 共现电影对列表 [(idx_i, idx_j), ...]
    这些 pair 来自同一用户喜欢的电影
    """
    model = AlignmentModel(
        text_dim=text_embeddings.shape[1],
        visual_dim=visual_embeddings.shape[1]
    ).cuda()
    optimizer = torch.optim.AdamW(model.parameters(), lr=1e-3, weight_decay=0.01)
    
    text_tensor = torch.tensor(text_embeddings, dtype=torch.float32)
    visual_tensor = torch.tensor(visual_embeddings, dtype=torch.float32)
    
    for epoch in range(epochs):
        np.random.shuffle(cooccurrence_pairs)
        total_loss = 0
        n_batches = 0
        
        for i in range(0, len(cooccurrence_pairs), batch_size):
            batch_pairs = cooccurrence_pairs[i:i+batch_size]
            idx_i = [p[0] for p in batch_pairs]
            idx_j = [p[1] for p in batch_pairs]
            
            # 电影 i 的融合表示
            fused_i = model.get_fused(
                text_tensor[idx_i].cuda(),
                visual_tensor[idx_i].cuda()
            )
            # 电影 j 的融合表示
            fused_j = model.get_fused(
                text_tensor[idx_j].cuda(),
                visual_tensor[idx_j].cuda()
            )
            
            loss = info_nce_loss(fused_i, fused_j, model.temperature)
            
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            
            total_loss += loss.item()
            n_batches += 1
        
        if (epoch + 1) % 10 == 0:
            print(f"Epoch {epoch+1}/{epochs}, Loss: {total_loss/n_batches:.4f}")
    
    # 生成所有电影的对齐后表示
    with torch.no_grad():
        all_fused = model.get_fused(
            text_tensor.cuda(), visual_tensor.cuda()
        ).cpu().numpy()
    
    return model, all_fused
```

# src/sid/explicit_tokens.py
"""
从 LLM 提取的语义标签 → 离散 token
"""
import json
from collections import Counter

# 预定义各维度的词表（根据实际数据分布调整）
EXPLICIT_VOCAB = {
    "mood": ["dark", "tense", "uplifting", "melancholic", "whimsical", 
             "intense", "serene", "humorous", "dramatic", "mysterious"],
    "pace": ["fast", "moderate", "slow", "varied"],
    "audience": ["general", "family", "mature", "cinephile", "teen"],
    "narrative_style": ["linear", "nonlinear", "ensemble", "character-study", 
                         "documentary-style", "anthology"],
    "visual_style": ["realistic", "stylized", "gritty", "colorful", 
                      "minimalist", "epic"],
}

def build_genre_vocab(tags_path: str, top_k: int = 40):
    """从数据中提取 top-K 高频细粒度类型"""
    with open(tags_path, 'r') as f:
        tags = json.load(f)
    
    counter = Counter()
    for t in tags.values():
        for g in t.get("genre_fine", []):
            counter[g.lower().strip()] += 1
    
    genre_vocab = [g for g, _ in counter.most_common(top_k)]
    return genre_vocab

def movie_to_explicit_tokens(movie_tags: dict, genre_vocab: list) -> list:
    """
    将一部电影的语义标签转为显式 token 序列
    输出示例：['sci-fi', 'thriller', 'dark', 'fast', 'mature', 'nonlinear']
    """
    tokens = []
    
    # 题材 token（最多 2 个）
    for g in movie_tags.get("genre_fine", [])[:2]:
        g = g.lower().strip()
        if g in genre_vocab:
            tokens.append(f"G_{g}")
        else:
            # 找最接近的，或标记为 OTHER
            tokens.append("G_other")
    
    # 情绪 token
    mood = movie_tags.get("mood", "").lower()
    if mood in EXPLICIT_VOCAB["mood"]:
        tokens.append(f"M_{mood}")
    
    # 节奏 token
    pace = movie_tags.get("pace", "").lower()
    if pace in EXPLICIT_VOCAB["pace"]:
        tokens.append(f"P_{pace}")
    
    # 受众 token
    aud = movie_tags.get("audience", "").lower()
    if aud in EXPLICIT_VOCAB["audience"]:
        tokens.append(f"A_{aud}")
    
    # 叙事风格 token
    ns = movie_tags.get("narrative_style", "").lower()
    if ns in EXPLICIT_VOCAB["narrative_style"]:
        tokens.append(f"N_{ns}")
    
    return tokens  # 长度约 4-6
```

### 5.2 RQ-VAE 隐式量化

```python
# src/sid/rqvae.py
"""
Residual Quantization: 将连续 embedding 量化为多层离散 code
比层次 K-Means 更优雅，梯度可训练

简化版实现用层次 K-Means 也可以，这里给出两个版本
"""
import torch
import torch.nn as nn
import numpy as np
from sklearn.cluster import MiniBatchKMeans

# ========== 方案 A：层次 K-Means（简单稳定，推荐先用这个）==========

def hierarchical_kmeans(embeddings: np.ndarray, n_levels: int = 3, 
                        k: int = 256, seed: int = 42) -> np.ndarray:
    """
    层次 K-Means 量化
    
    Args:
        embeddings: [N, D] 融合后的多模态 embedding
        n_levels: 量化层数
        k: 每层的聚类数
    
    Returns:
        codes: [N, n_levels] 每部电影的隐式 token 序列
    """
    codes = np.zeros((len(embeddings), n_levels), dtype=np.int32)
    residual = embeddings.copy()
    
    codebooks = []
    for level in range(n_levels):
        kmeans = MiniBatchKMeans(n_clusters=k, random_state=seed + level, 
                                 batch_size=1024, n_init=3)
        kmeans.fit(residual)
        
        codes[:, level] = kmeans.labels_
        centers = kmeans.cluster_centers_[kmeans.labels_]
        residual = residual - centers  # 残差
        
        codebooks.append(kmeans)
        print(f"Level {level}: inertia={kmeans.inertia_:.2f}")
    
    return codes, codebooks


# ========== 方案 B：RQ-VAE（可训练，效果上限更高）==========

class ResidualVQ(nn.Module):
    """可训练的残差向量量化"""
    def __init__(self, dim: int, n_levels: int = 3, codebook_size: int = 256):
        super().__init__()
        self.n_levels = n_levels
        self.codebook_size = codebook_size
        
        # 每层一个 codebook
        self.codebooks = nn.ParameterList([
            nn.Parameter(torch.randn(codebook_size, dim) * 0.01)
            for _ in range(n_levels)
        ])
    
    def quantize_one_level(self, x, codebook):
        # x: [B, D], codebook: [K, D]
        dist = torch.cdist(x, codebook)  # [B, K]
        indices = dist.argmin(dim=-1)    # [B]
        quantized = codebook[indices]     # [B, D]
        return quantized, indices
    
    def forward(self, x):
        """
        x: [B, D]
        returns: quantized [B, D], indices [B, n_levels], commitment_loss
        """
        residual = x
        all_indices = []
        all_quantized = []
        commitment_loss = 0
        
        for level in range(self.n_levels):
            quantized, indices = self.quantize_one_level(residual, self.codebooks[level])
            
            # Straight-through estimator
            quantized_st = residual + (quantized - residual).detach()
            
            commitment_loss += F.mse_loss(residual, quantized.detach())
            
            residual = residual - quantized_st
            all_indices.append(indices)
            all_quantized.append(quantized_st)
        
        total_quantized = sum(all_quantized)
        all_indices = torch.stack(all_indices, dim=-1)  # [B, n_levels]
        
        return total_quantized, all_indices, commitment_loss
```

# src/sid/build_sid.py
"""
将显式 token + 隐式 token 拼接为最终的 Semantic ID
"""
import json
import numpy as np
from .explicit_tokens import movie_to_explicit_tokens, build_genre_vocab
from .rqvae import hierarchical_kmeans

def build_all_sids(tags_path: str, fused_embeddings: np.ndarray, 
                   movie_ids: list, output_path: str):
    """
    构建所有电影的 Semantic ID
    
    最终 SID 形式示例：
    ["G_sci-fi", "G_thriller", "M_dark", "P_fast", "A_mature", "N_nonlinear", "Q0_17", "Q1_52", "Q2_8"]
    """
    with open(tags_path, 'r') as f:
        all_tags = json.load(f)
    
    genre_vocab = build_genre_vocab(tags_path)
    
    # 隐式量化
    codes, codebooks = hierarchical_kmeans(fused_embeddings, n_levels=3, k=256)
    
    # 构建全局 token 词表
    token_vocab = {"[PAD]": 0, "[BOS]": 1, "[EOS]": 2, "[SEP]": 3}
    
    sid_map = {}
    for i, mid in enumerate(movie_ids):
        tags = all_tags.get(mid, {})
        
        # 显式 tokens
        explicit = movie_to_explicit_tokens(tags, genre_vocab)
        
        # 隐式 tokens
        implicit = [f"Q{level}_{codes[i, level]}" for level in range(3)]
        
        # 完整 SID
        sid = explicit + implicit
        
        # 注册到词表
        for token in sid:
            if token not in token_vocab:
                token_vocab[token] = len(token_vocab)
        
        sid_map[mid] = {
            "sid_tokens": sid,
            "sid_ids": [token_vocab[t] for t in sid],
            "explicit_len": len(explicit),
            "implicit_len": len(implicit),
        }
    
    result = {
        "vocab": token_vocab,
        "vocab_size": len(token_vocab),
        "sids": sid_map,
    }
    
    with open(output_path, 'w', encoding='utf-8') as f:
        json.dump(result, f, ensure_ascii=False, indent=2)
    
    print(f"SID 构建完成：{len(sid_map)} 部电影，词表大小 {len(token_vocab)}")
    
    # 统计
    sid_lens = [len(s["sid_tokens"]) for s in sid_map.values()]
    print(f"SID 长度统计：min={min(sid_lens)}, max={max(sid_lens)}, "
          f"avg={np.mean(sid_lens):.1f}")
    
    return result

```python
# src/model/transformer_rec.py
"""
TIGER 风格的生成式推荐模型
输入：用户历史电影的 SID 序列（拼接）
输出：自回归预测下一部电影的 SID token 序列
"""
import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class SIDGenerativeRecommender(nn.Module):
    def __init__(self, vocab_size: int, d_model: int = 256, n_heads: int = 8,
                 n_layers: int = 6, max_seq_len: int = 512, dropout: float = 0.1):
        super().__init__()
        self.d_model = d_model
        self.max_seq_len = max_seq_len
        
        # Token embedding + 位置编码
        self.token_emb = nn.Embedding(vocab_size, d_model)
        self.pos_emb = nn.Embedding(max_seq_len, d_model)
        self.dropout = nn.Dropout(dropout)
        
        # Transformer Decoder
        decoder_layer = nn.TransformerDecoderLayer(
            d_model=d_model, nhead=n_heads, dim_feedforward=d_model * 4,
            dropout=dropout, activation='gelu', batch_first=True
        )
        self.decoder = nn.TransformerDecoder(decoder_layer, num_layers=n_layers)
        
        # 输出投影
        self.output_proj = nn.Linear(d_model, vocab_size)
        
        # 因果 mask
        self.register_buffer("causal_mask", None)
        
        self._init_weights()
    
    def _init_weights(self):
        nn.init.normal_(self.token_emb.weight, std=0.02)
        nn.init.normal_(self.pos_emb.weight, std=0.02)
    
    def _get_causal_mask(self, seq_len: int, device):
        if self.causal_mask is None or self.causal_mask.shape[0] < seq_len:
            mask = torch.triu(torch.ones(seq_len, seq_len, device=device), diagonal=1).bool()
            self.causal_mask = mask
        return self.causal_mask[:seq_len, :seq_len]
    
    def forward(self, input_ids: torch.Tensor, labels: torch.Tensor = None):
        """
        input_ids: [B, L] — 用户历史 SID 拼接序列
        labels:    [B, L] — shifted 目标序列
        """
        B, L = input_ids.shape
        positions = torch.arange(L, device=input_ids.device).unsqueeze(0)
        
        x = self.token_emb(input_ids) + self.pos_emb(positions)
        x = self.dropout(x)
        
        # 因果 mask
        causal_mask = self._get_causal_mask(L, input_ids.device)
        
        # 用 decoder 的 self-attention（没有 encoder，类似 GPT）
        memory = torch.zeros(B, 1, self.d_model, device=input_ids.device)
        x = self.decoder(x, memory, tgt_mask=causal_mask)
        
        logits = self.output_proj(x)  # [B, L, vocab_size]
        
        loss = None
        if labels is not None:
            loss = F.cross_entropy(
                logits.view(-1, logits.size(-1)),
                labels.view(-1),
                ignore_index=0  # PAD token
            )
        
        return {"logits": logits, "loss": loss}
    
    @torch.no_grad()
    def generate(self, input_ids: torch.Tensor, max_new_tokens: int = 9,
                 temperature: float = 0.8, top_k: int = 50):
        """
        自回归生成下一部电影的 SID
        max_new_tokens: SID 长度（显式 ~6 + 隐式 3 = ~9）
        """
        self.eval()
        for _ in range(max_new_tokens):
            # 截断到 max_seq_len
            input_crop = input_ids[:, -self.max_seq_len:]
            output = self.forward(input_crop)
            logits = output["logits"][:, -1, :]  # 最后一个位置
            
            # Temperature + Top-K 采样
            logits = logits / temperature
            if top_k > 0:
                v, _ = torch.topk(logits, top_k)
                logits[logits < v[:, [-1]]] = float('-inf')
            
            probs = F.softmax(logits, dim=-1)
            next_token = torch.multinomial(probs, num_samples=1)
            
            input_ids = torch.cat([input_ids, next_token], dim=-1)
            
            # 遇到 EOS 停止
            if (next_token == 2).all():  # EOS token id = 2
                break
        
        return input_ids
```

### 6.2 数据集构造

```python
# src/model/dataset.py
"""
将用户的观影序列转换为 SID token 序列
"""
import torch
from torch.utils.data import Dataset
import json

class SIDSequenceDataset(Dataset):
    def __init__(self, user_sequences: list, sid_data: dict, max_len: int = 512):
        """
        user_sequences: [{"userId": ..., "movie_seq": [m1, m2, ...], "target": m_target}, ...]
        sid_data: build_sid.py 的输出
        """
        self.data = []
        self.max_len = max_len
        sid_map = sid_data["sids"]
        
        BOS = sid_data["vocab"]["[BOS]"]
        EOS = sid_data["vocab"]["[EOS]"]
        SEP = sid_data["vocab"]["[SEP]"]
        PAD = sid_data["vocab"]["[PAD]"]
        
        for item in user_sequences:
            # 将历史序列的每部电影 SID 拼接
            tokens = [BOS]
            for mid in item["movie_seq"]:
                mid_str = str(mid)
                if mid_str in sid_map:
                    tokens.extend(sid_map[mid_str]["sid_ids"])
                    tokens.append(SEP)
            
            # 目标电影的 SID
            if "target" in item:
                target_mid = str(item["target"])
                if target_mid in sid_map:
                    target_tokens = sid_map[target_mid]["sid_ids"] + [EOS]
                else:
                    continue
            else:
                continue
            
            # 完整序列 = 历史 + 目标
            full_seq = tokens + target_tokens
            
            # 截断
            if len(full_seq) > max_len:
                # 保留最近的历史 + 完整目标
                full_seq = full_seq[-(max_len):]
            
            self.data.append({
                "input_ids": full_seq[:-1],
                "labels": full_seq[1:],
                "target_sid": sid_map.get(str(item.get("target", "")), {}).get("sid_ids", []),
            })
    
    def __len__(self):
        return len(self.data)
    
    def __getitem__(self, idx):
        item = self.data[idx]
        return {
            "input_ids": torch.tensor(item["input_ids"], dtype=torch.long),
            "labels": torch.tensor(item["labels"], dtype=torch.long),
        }

def collate_fn(batch, pad_id=0):
    """动态 padding"""
    max_len = max(len(b["input_ids"]) for b in batch)
    input_ids = torch.full((len(batch), max_len), pad_id, dtype=torch.long)
    labels = torch.full((len(batch), max_len), pad_id, dtype=torch.long)
    
    for i, b in enumerate(batch):
        L = len(b["input_ids"])
        input_ids[i, :L] = b["input_ids"]
        labels[i, :L] = b["labels"]
    
    return {"input_ids": input_ids, "labels": labels}
```

### 6.3 训练

```python
# src/model/train.py
import torch
from torch.utils.data import DataLoader
from torch.optim import AdamW
from torch.optim.lr_scheduler import CosineAnnealingLR
import json
from tqdm import tqdm

from .transformer_rec import SIDGenerativeRecommender
from .dataset import SIDSequenceDataset, collate_fn

def train(config: dict):
    # 加载数据
    with open(config["sid_path"], 'r') as f:
        sid_data = json.load(f)
    
    with open(config["train_path"], 'r') as f:
        train_seqs = json.load(f)
    with open(config["val_path"], 'r') as f:
        val_seqs = json.load(f)
    
    train_dataset = SIDSequenceDataset(train_seqs, sid_data, max_len=config["max_len"])
    val_dataset = SIDSequenceDataset(val_seqs, sid_data, max_len=config["max_len"])
    
    train_loader = DataLoader(train_dataset, batch_size=config["batch_size"],
                               shuffle=True, collate_fn=collate_fn, num_workers=4)
    val_loader = DataLoader(val_dataset, batch_size=config["batch_size"],
                             shuffle=False, collate_fn=collate_fn, num_workers=4)
    
    # 模型
    model = SIDGenerativeRecommender(
        vocab_size=sid_data["vocab_size"],
        d_model=config.get("d_model", 256),
        n_heads=config.get("n_heads", 8),
        n_layers=config.get("n_layers", 6),
        max_seq_len=config.get("max_len", 512),
    ).cuda()
    
    optimizer = AdamW(model.parameters(), lr=config.get("lr", 1e-3),
                      weight_decay=config.get("wd", 0.01))
    scheduler = CosineAnnealingLR(optimizer, T_max=config["epochs"])
    
    best_val_loss = float('inf')
    
    for epoch in range(config["epochs"]):
        # Train
        model.train()
        total_loss = 0
        for batch in tqdm(train_loader, desc=f"Epoch {epoch+1}"):
            input_ids = batch["input_ids"].cuda()
            labels = batch["labels"].cuda()
            
            output = model(input_ids, labels)
            loss = output["loss"]
            
            optimizer.zero_grad()
            loss.backward()
            torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            optimizer.step()
            
            total_loss += loss.item()
        
        avg_train_loss = total_loss / len(train_loader)
        
        # Validate
        model.eval()
        val_loss = 0
        with torch.no_grad():
            for batch in val_loader:
                input_ids = batch["input_ids"].cuda()
                labels = batch["labels"].cuda()
                output = model(input_ids, labels)
                val_loss += output["loss"].item()
        
        avg_val_loss = val_loss / len(val_loader)
        scheduler.step()
        
        print(f"Epoch {epoch+1}: train_loss={avg_train_loss:.4f}, val_loss={avg_val_loss:.4f}")
        
        if avg_val_loss < best_val_loss:
            best_val_loss = avg_val_loss
            torch.save(model.state_dict(), config["save_path"])
            print(f"  → 模型已保存")
```

### 6.4 评测：SID → 电影映射 + 指标计算

```python
# src/model/evaluate.py
"""
推理流程：
1. 输入用户历史 SID 序列
2. 自回归生成一个预测 SID
3. 将预测 SID 与所有电影的 SID 做匹配
4. 返回 top-K 候选电影
"""
import torch
import numpy as np
from collections import Counter

def sid_to_str(sid_tokens: list) -> str:
    return "|".join(str(t) for t in sid_tokens)

def compute_sid_similarity(pred_sid: list, candidate_sid: list) -> float:
    """
    SID 相似度：加权匹配
    显式 token 完全匹配 + 隐式 token 距离
    """
    # 简单版本：token overlap + 位置加权
    score = 0
    min_len = min(len(pred_sid), len(candidate_sid))
    
    for i in range(min_len):
        if pred_sid[i] == candidate_sid[i]:
            # 前面的 token（更高层语义）权重更大
            score += 1.0 / (i + 1)
    
    return score

def beam_search_match(model, input_ids: torch.Tensor, all_sids: dict,
                      beam_size: int = 20, sid_len: int = 9):
    """
    Beam search 生成 + 候选匹配
    返回 top-K 电影列表
    """
    # 生成多个候选 SID
    candidates = []
    for _ in range(beam_size):
        generated = model.generate(input_ids.clone(), max_new_tokens=sid_len,
                                    temperature=0.9, top_k=50)
        new_tokens = generated[0, input_ids.shape[1]:].tolist()
        candidates.append(new_tokens)
    
    # 与所有电影 SID 匹配
    movie_scores = {}
    for mid, sid_info in all_sids.items():
        movie_sid = sid_info["sid_ids"]
        best_score = 0
        for cand in candidates:
            score = compute_sid_similarity(cand, movie_sid)
            best_score = max(best_score, score)
        movie_scores[mid] = best_score
    
    # 排序返回 top-K
    ranked = sorted(movie_scores.items(), key=lambda x: x[1], reverse=True)
    return ranked

def evaluate(model, test_loader, sid_data, top_k_list=[5, 10, 20]):
    """计算 Recall@K, NDCG@K, HitRate@K, MRR"""
    model.eval()
    all_sids = sid_data["sids"]
    
    metrics = {f"Recall@{k}": [] for k in top_k_list}
    metrics.update({f"NDCG@{k}": [] for k in top_k_list})
    metrics["MRR"] = []
    
    for batch in test_loader:
        input_ids = batch["input_ids"].cuda()
        target_mids = batch["target_mid"]
        
        for i in range(input_ids.shape[0]):
            ranked = beam_search_match(model, input_ids[i:i+1], all_sids)
            ranked_mids = [mid for mid, _ in ranked]
            target = target_mids[i]
            
            # 计算指标
            for k in top_k_list:
                top_k_mids = ranked_mids[:k]
                hit = 1 if target in top_k_mids else 0
                metrics[f"Recall@{k}"].append(hit)
                
                # NDCG
                if target in top_k_mids:
                    rank = top_k_mids.index(target) + 1
                    metrics[f"NDCG@{k}"].append(1.0 / np.log2(rank + 1))
                else:
                    metrics[f"NDCG@{k}"].append(0)
            
            # MRR
            if target in ranked_mids:
                rank = ranked_mids.index(target) + 1
                metrics["MRR"].append(1.0 / rank)
            else:
                metrics["MRR"].append(0)
    
    # 汇总
    results = {k: np.mean(v) for k, v in metrics.items()}
    return results
```
